#!/usr/bin/env bash

# Script para utilizar no meu autostart para atualizar automáticamente
# toda vez que iniciar - APLICAÇÃO PESSOAL

sleep 5
var=$(xdotool getactivewindow)
xdotool windowminimize "$var"

echo -e "9124\n" | sudo -S atualizar.sh
echo ""
echo ">>> Finalizado !"
echo ""
sleep 5
