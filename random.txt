instalr o wine

1) $ sudo dpkg --add-architecture i386 // (ativando arquiteturo 32bits)
2) $ wget -nc https://dl.winehq.org/wine-builds/winehq.key // (download da chave do repo)
3) $ sudo apt-key add winehq.key // (adicionar chave do repo)
4) $ sudo add-apt-repository 'deb https://dl.winehq.org/wine-builds/ubuntu/ focal main'
5) $ sudo add-apt-repository ppa:cybermax-dexter/sdl2-backport // para add ppa pelo terminal
